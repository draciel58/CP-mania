void dfs(int node,vector<vector<int>>& friends,vector<int>& visited){
    visited[node]=1;
    for(int j:friends[node]){
        if(visited[j] == 0){
            dfs(j,friends,visited);
        }
    }
}

int solve(vector<vector<int>>& friends) {
    int n = friends.size();
    vector<int> visited(n,0);
    int ans=0;
    for(int i=0;i<n;i++){
        if(visited[i]==0){
            dfs(i,friends,visited);
            ans++;
        }
    }
    return ans;
}
