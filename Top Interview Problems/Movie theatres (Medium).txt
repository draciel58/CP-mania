int solve(vector<vector<int>>& intervals) {
    int n = intervals.size();
    map<int,int> mp;
    for(int i=0;i<n;i++){
        mp[intervals[i][0]]++;
        mp[intervals[i][1]]--;
    }
    int ans=0,temp=0;
    for(auto it:mp){
        temp+=it.second;
        ans=max(ans,temp);
    }
    return ans;
}
