bool solve(string s, string t) {
    
    /*sort(s.begin(),s.end());
    sort(t.begin(),t.end());
    if(s==t) return true;
    return false;
    */

    int n = s.size();
    int m = t.size();
    map<char,int> mp;
    for(int i=0;i<n;i++){
        mp[s[i]]++;
    }
    for(int i=0;i<m;i++){
        mp[t[i]]--;
    }
    for(auto it:mp){
        if(it.second != 0) return false;
    }
    return true;
}
